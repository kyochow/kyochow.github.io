lua chunk block encode

该方案会直接修改lua库中luac和unluac相关的逻辑，这种方式会比文件加密有更复杂的加密逻辑，也急剧增加破解难度

核心方法

  • ldump.c中的DumpBlock
  • lundump.c中的LoadBlock

他俩是成对的，如何的加密，就要对应的解密


static void DumpBlock(const void* b, size_t size, DumpState* D)
{
     if (D->status==0)
     {
          lua_unlock(D->L);
          D->status=(*D->writer)(D->L,b,size,D->data);
          lua_lock(D->L);
     }
}


static void LoadBlock(LoadState* S, void* b, size_t size)
{
     size_t r=luaZ_read(S->Z,b,size);
     IF (r!=0, "unexpected end");
}

如何写呢？这里给出一个示例

static void DumpBlock(const void* b, size_t size, DumpState* D)
{
     if (D->status==0)
     {
          BlockEncrypt(b,size);
          lua_unlock(D->L);
          D->status=(*D->writer)(D->L,b,size,D->data);
          lua_lock(D->L);
          BlockDecrypt(b,size);
     }
}

static void LoadBlock(LoadState* S, void* b, size_t size)
{
     size_t r=luaZ_read(S->Z,b,size);
     BlockDecrypt(b,size);
     IF (r!=0, "unexpected end");
}

BlockEncrypt(b,s)	
{
    size_t i;
    for(i=1;i<s;++i)
    {
        cast(char*,b)[i]=cast(char*,b)[i]^XXX;//XXX代表要异或的char
    }
}

BlockDecrypt(b,s)	
{
    size_t i;
    for(i=(s>0?s-1:0);i>0;--i)
    {
        cast(char*,b)[i]=cast(char*,b)[i]^XXX;//XXX代表要异或的char
    }
}
